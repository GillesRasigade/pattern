Example
class Person extends EventSourcing {
  constructor(data = { name: 'unknown' }) {
    super(data);
  }
  setName(name) {
    this.push('setName', name);
    this.data.name = name;
  }
}

const alice = new Person();
alice.setName('Alice');

console.log(alice);
