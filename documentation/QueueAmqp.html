Example
const QueueAmqp = require('QueueAmqp');

const queue = new QueueAmqp({ queue: 'tasks', rx: true, tx: true });

queue
.connect()
.then(() => {
  queue.on('task', message => {
    console.log(message);
  });
  queue.emit('task', 'A coffee please !');
});
