Example
const a = { sum: 0 };

const command = new Command(
  (subject, inc) => subject.sum += inc,
  (subject, inc) => subject.sum -= inc,
  [a, 1]
);

command.execute();
// a.sum = 1

command.undo();
// a.sum = 0
